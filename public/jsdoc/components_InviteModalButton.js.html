// Invite Friends系のコンポ
///////////////////////////////////////////////////////////////////////////////////////////////
import Modal from "react-bootstrap/Modal"
import React, { PureComponent, useEffect, useState } from "react"
import Button from "react-bootstrap/Button"
import Form from "react-bootstrap/Form"

/**
 * 【InviteModalButton】押すとメールアドレスを入れるフォームが開く
 * 〇InviteModalButton
 *  ・InviteMailForm
 *  ・InviteSubmitButton
 * 
If you succeed in sending invitation, this window will close.
 */
export default function InviteModalButton(inviteSuccess) {
  const [show, setShow] = useState(false)
  const handleShow = () => setShow(true)
  const handleClose = () => setShow(false)

  let modalBody
  if (inviteSuccess) {
    //trueのときにinvitedが出るはずなんだけども・・・・？
    modalBody = (
      <Modal.Body>
        {" "}
        <InviteMailForm />
        <InviteSubmitButton />
      </Modal.Body>
    )
  } else {
    modalBody = <Modal.Body>You invited friend!</Modal.Body>
  }

  return (
    <>
      <Button variant="info" onClick={handleShow}>
        Invite your friends!!!
      </Button>

      <Modal show={show} onHide={handleClose}>
        {modalBody}
        <Modal.Footer>
          <Button variant="secondary" onClick={handleClose}>
            Close
          </Button>
        </Modal.Footer>
      </Modal>
    </>
  )
}

export function InviteMailForm() {
  return (
    <Form>
      <Form.Group controlId="formBasicEmail">
        <Form.Label>Friends email address </Form.Label>
        <Form.Control type="email" placeholder="Enter email" />
      </Form.Group>
    </Form>
  )
}

export function InviteSubmitButton() {
  return (
    //onClickは後ほど入れる
    <Button type="submit">Invite💌</Button>
  )
}
