export function CatPage({
  catPageNum,
  maxPageNum,
  currentPage,
  CatSearch,
  searchBreeds,
}) {
  let prev =
    currentPage <= 1 ? (
      ""

      <CatPagePrev

        CatSearch={CatSearch}

      />
    )
  let next =
    currentPage === 0 || currentPage === maxPageNum ? (
      ""
    ) : (
      <CatPageNext
        currentPage={currentPage}
        CatSearch={CatSearch}
        searchBreeds={searchBreeds}
      />
    )

  return (
    <table className="table" align="center">
      <tbody>
        <tr>
          <td>{prev}</td>

          <td>
            {
              catPageNum.map((page) => (
                <CatPageNum
                  key={page}
                  catPageNum={page}
                  CatSearch={CatSearch}
                  searchBreeds={searchBreeds}
                />
              )) /* å®Œäº† */
            }
          </td>

}
